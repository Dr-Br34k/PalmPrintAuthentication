329  Function

332  public:
333  // `Boundary` needs to be defaulted to `Count` so that the `run` function
334  // template is not implicitly instantiated on class template instantiation.
335  // Look up section 14.7.1 _Implicit instantiation_ of the ISO C++14 Standard
336  // and the [dicussion](https://github.com/aminroosta/sqlite_modern_cpp/issues/8)
337  // on Github.
338 
339  template<
340  typename Function,
341  typename... Values,
342  std::size_t Boundary = Count
343  >
344  static typename std::enable_if<(sizeof...(Values) < Boundary), void>::type run(
345  database_binder& db,
346  Function& function,
347  Values&&... values
348  ) {
349  nth_argument_type<Function, sizeof...(Values)> value{};
350  get_col_from_db(db, sizeof...(Values), value);
351 
352  run<Function>(db, function, std::forward<Values>(values)..., std::move(value));
353  }

361  database_binder&,
362  Function& function,
363  Values&&... values
364  ) {
365  function(std::move(values)...);
366  }
